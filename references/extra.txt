# 1️⃣ Data Ingestion
dvc stage add -n data_ingestion \
    -d src/data/data_ingestion.py -d params.yaml \
    -o data/raw \
    python src/data/data_ingestion.py

# 2️⃣ Data Preprocessing
dvc stage add -n data_preprocessing \
    -d src/data/data_preprocessing.py -d data/raw -d params.yaml \
    -o data/interim \
    python src/data/data_preprocessing.py

# 3️⃣ Feature Engineering
dvc stage add -n feature_engineering \
    -d src/features/feature_engineering.py -d data/interim -d params.yaml \
    -o data/processed \
    python src/features/feature_engineering.py

# 4️⃣ Model Building
dvc stage add -n model_building \
    -d src/models/model_building.py -d data/processed -d params.yaml \
    -o models/model.pkl \
    python src/models/model_building.py

# 5️⃣ Model Evaluation
dvc stage add -n model_evaluation \
    -d src/models/model_evaluation.py -d models/model.pkl # -d data/processed -d params.yaml \
    -o reports/metrics.json -o reports/confusion_matrix.png \
    python src/models/model_evaluation.py
